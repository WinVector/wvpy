  9def convert_py_code_to_notebook(text: str) -> nbformat.notebooknode.NotebookNode:
 10    """
 11    Convert python text to a notebook. 
 12    "''' begin text" ends any open blocks, and starts a new markdown block
 13    "''' # end text" ends text, and starts a new code block
 14    "'''end code'''" ends code blocks, and starts a new code block
 15
 16    :param text: Python text to convert.
 17    :return: a notebook 
 18    """
 19    assert isinstance(text, str)
 20    lines = text.splitlines()
 21    begin_text_regexp = re.compile(r"^r?'''\s*begin\s+text\s*$")
 22    end_text_regexp = re.compile(r"^'''\s*#\s*end\s+text\s*$")
 23    end_code_regexp = re.compile(r"^r?'''\s*end\s+code\s*'''\s*$")
 24    nbf_v = nbformat.v4
 25    nb = nbf_v.new_notebook()
 26    # run a little code collecting state machine
 27    cells = []
 28    collecting_python = []
 29    collecting_text = None
 30    lines.append(None)  # append an ending sentinel
 31    # scan input
 32    for line in lines:
 33        if line is None:
 34            is_end = True
 35            text_start = False
 36            code_start = False
 37            code_end = False
 38        else:
 39            is_end = False
 40            text_start = begin_text_regexp.match(line)
 41            code_start = end_text_regexp.match(line)
 42            code_end = end_code_regexp.match(line)
 43        if is_end or text_start or code_start or code_end:
 44            if (collecting_python is not None) and (len(collecting_python) > 0):
 45                txt_block = ('\n'.join(collecting_python)).strip('\n') + '\n'
 46                if len(txt_block.strip()) > 0:
 47                    cells.append(nbf_v.new_code_cell(txt_block))
 48            if (collecting_text is not None) and (len(collecting_text) > 0):
 49                txt_block = ('\n'.join(collecting_text)).strip('\n') + '\n'
 50                if len(txt_block.strip()) > 0:
 51                    cells.append(nbf_v.new_markdown_cell(txt_block))
 52            collecting_python = None
 53            collecting_text = None
 54            if not is_end:
 55                if text_start:
 56                    collecting_text = []
 57                else:
 58                    collecting_python = []
 59        else:
 60            if collecting_python is not None:
 61                collecting_python.append(line)
 62            if collecting_text is not None:
 63                collecting_text.append(line)
 64    nb['cells'] = cells
 65    return nb

 89def convert_notebook_code_to_py(nb: nbformat.notebooknode.NotebookNode) -> str:
 90    """
 91    Convert ipython notebook inputs to a py code. 
 92    "''' begin text" ends any open blocks, and starts a new markdown block
 93    "''' # end text" ends text, and starts a new code block
 94    "'''end code'''" ends code blocks, and starts a new code block
 95
 96    :param nb: notebook
 97    :return: Python source code
 98    """
 99    res = []
100    code_needs_end = False
101    for cell in nb.cells:
102        if len(cell.source.strip()) > 0:
103            if cell.cell_type == 'code':
104                if code_needs_end:
105                    res.append("\n'''end code'''\n")
106                res.append(cell.source.strip('\n'))
107                code_needs_end = True
108            else:
109                res.append("\n''' begin text")
110                res.append(cell.source.strip('\n'))
111                res.append("'''  # end text\n")
112                code_needs_end = False
113    res_text = '\n' + ('\n'.join(res)).strip('\n') + '\n\n'
114    return res_text

142def render_as_html(
143    notebook_file_name: str,
144    *,
145    output_suffix: Optional[str] = None,
146    timeout:int = 60000,
147    kernel_name: Optional[str] = None,
148    verbose: bool = True,
149    parameters = None,
150    parameter_file_name: Optional[str] = None,
151    exclude_input: bool = False,
152    prompt_strip_regexp: Optional[str] = r'<\s*div\s+class\s*=\s*"jp-OutputPrompt[^<>]*>[^<>]*Out[^<>]*<\s*/div\s*>',
153) -> None:
154    """
155    Render a Jupyter notebook in the current directory as HTML.
156    Exceptions raised in the rendering notebook are allowed to pass trough.
157
158    :param notebook_file_name: name of source file, must end with .ipynb or .py
159    :param output_suffix: optional name to add to result name
160    :param timeout: Maximum time in seconds each notebook cell is allowed to run.
161                    passed to nbconvert.preprocessors.ExecutePreprocessor.
162    :param kernel_name: Jupyter kernel to use. passed to nbconvert.preprocessors.ExecutePreprocessor.
163    :param verbose logical, if True print while running 
164    :param parameters: arbitrary object to write to paramater_file_name
165    :param parameter_file_name: file name to pickle parameters to
166    :param exclude_input: if True, exclude input cells
167    :param prompt_strip_regexp: regexp to strip prompts, only used if exclude_input is True
168    :return: None
169    """
170    assert isinstance(notebook_file_name, str)
171    if notebook_file_name.endswith(".ipynb"):
172        suffix = ".ipynb"
173        with open(notebook_file_name, "rb") as f:
174            nb = nbformat.read(f, as_version=4)
175    elif notebook_file_name.endswith(".py"):
176        suffix = ".py"
177        with open(notebook_file_name, 'r') as f:
178            text = f.read()
179        nb = convert_py_code_to_notebook(text)
180    else:
181        raise ValueError("file name must end with .ipynb or .py")
182    html_name = notebook_file_name.removesuffix(suffix)
183    exec_note = ""
184    if output_suffix is not None:
185        assert isinstance(output_suffix, str)
186        html_name = html_name + output_suffix
187        exec_note = f'"{output_suffix}"'
188    html_name = html_name + ".html"
189    try:
190        os.remove(html_name)
191    except FileNotFoundError:
192        pass
193    if parameter_file_name is not None:
194        assert isinstance(parameter_file_name, str)
195        with open(parameter_file_name, "wb") as f:
196            pickle.dump(parameters, f)
197    caught = None
198    try:
199        if verbose:
200            print(
201                f'start render_as_html "{notebook_file_name}" {exec_note} {datetime.datetime.now()}'
202            )
203        if kernel_name is not None:
204            ep = nbconvert.preprocessors.ExecutePreprocessor(
205                timeout=timeout, kernel_name=kernel_name
206            )
207        else:
208            ep = nbconvert.preprocessors.ExecutePreprocessor(timeout=timeout)
209        nb_res, nb_resources = ep.preprocess(nb)
210        html_exporter = nbconvert.HTMLExporter(exclude_input=exclude_input)
211        html_body, html_resources = html_exporter.from_notebook_node(nb_res)
212        if exclude_input and (prompt_strip_regexp is not None):
213            # strip output prompts
214            html_body = re.sub(
215                prompt_strip_regexp,
216                ' ',
217                html_body)
218        with open(html_name, "wt") as f:
219            f.write(html_body)
220    except Exception as e:
221        caught = e
222    if parameter_file_name is not None:
223        try:
224            os.remove(parameter_file_name)
225        except FileNotFoundError:
226            pass
227    if caught is not None:
228        raise caught
229    if verbose:
230        print(f'\tdone render_as_html "{html_name}" {datetime.datetime.now()}')
