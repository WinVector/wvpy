 39def pretty_format_python(python_txt: str, *, black_mode=None) -> str:
 40    """
 41    Format Python code, using black.
 42
 43    :param python_txt: Python code
 44    :param black_mode: options for black
 45    :return: formatted Python code
 46    """
 47    assert have_black
 48    assert isinstance(python_txt, str)
 49    formatted_python = python_txt.strip("\n") + "\n"
 50    if len(formatted_python.strip()) > 0:
 51        if black_mode is None:
 52            black_mode = black.FileMode()
 53        try:
 54            formatted_python = black.format_str(formatted_python, mode=black_mode)
 55            formatted_python = formatted_python.strip("\n") + "\n"
 56        except Exception:
 57            pass
 58    return formatted_python

 61def convert_py_code_to_notebook(
 62    text: str, *, use_black: bool = False
 63) -> nbformat.notebooknode.NotebookNode:
 64    """
 65    Convert python text to a notebook.
 66    "''' begin text" ends any open blocks, and starts a new markdown block (triple double quotes also allowed)
 67    "''' # end text" ends text, and starts a new code block (triple double quotes also allowed)
 68    "'''end code'''" ends code blocks, and starts a new code block (triple double quotes also allowed)
 69
 70    :param text: Python text to convert.
 71    :param use_black: if True use black to re-format Python code
 72    :return: a notebook
 73    """
 74    # https://stackoverflow.com/a/23729611/6901725
 75    # https://nbviewer.org/gist/fperez/9716279
 76    assert isinstance(text, str)
 77    assert isinstance(use_black, bool)
 78    lines = text.splitlines()
 79    begin_text_regexp = re.compile(r"^\s*r?((''')|(\"\"\"))\s*begin\s+text\s*$")
 80    end_text_regexp = re.compile(r"^\s*r?((''')|(\"\"\"))\s*#\s*end\s+text\s*$")
 81    end_code_regexp = re.compile(
 82        r"(^\s*r?'''\s*end\s+code\s*'''\s*$)|(^\s*r?\"\"\"\s*end\s+code\s*\"\"\"\s*$)"
 83    )
 84    # run a little code collecting state machine
 85    cells = []
 86    collecting_python = []
 87    collecting_text = None
 88    lines.append(None)  # append an ending sentinel
 89    # scan input
 90    for line in lines:
 91        if line is None:
 92            is_end = True
 93            text_start = False
 94            code_start = False
 95            code_end = False
 96        else:
 97            is_end = False
 98            text_start = begin_text_regexp.match(line)
 99            code_start = end_text_regexp.match(line)
100            code_end = end_code_regexp.match(line)
101        if is_end or text_start or code_start or code_end:
102            if (collecting_python is not None) and (len(collecting_python) > 0):
103                python_block = ("\n".join(collecting_python)).strip("\n") + "\n"
104                if len(python_block.strip()) > 0:
105                    if use_black and have_black:
106                        python_block = pretty_format_python(python_block)
107                    cells.append(nbf_v.new_code_cell(python_block))
108            if (collecting_text is not None) and (len(collecting_text) > 0):
109                txt_block = ("\n".join(collecting_text)).strip("\n") + "\n"
110                if len(txt_block.strip()) > 0:
111                    cells.append(nbf_v.new_markdown_cell(txt_block))
112            collecting_python = None
113            collecting_text = None
114            if not is_end:
115                if text_start:
116                    collecting_text = []
117                else:
118                    collecting_python = []
119        else:
120            if collecting_python is not None:
121                collecting_python.append(line)
122            if collecting_text is not None:
123                collecting_text.append(line)
124    for i in range(len(cells)):
125        cells[i]["id"] = f"cell{i}"
126    nb = nbf_v.new_notebook(cells=cells)
127    # nb = _normalize(nb)
128    return nb

131def prepend_code_cell_to_notebook(
132    nb: nbformat.notebooknode.NotebookNode,
133    *,
134    code_text: str,
135) -> nbformat.notebooknode.NotebookNode:
136    """
137    Prepend a code cell to a Jupyter notebook.
138
139    :param nb: Jupyter notebook to alter
140    :param code_text: Python source code to add
141    :return: new notebook
142    """
143    header_cell = nbf_v.new_code_cell(code_text)
144    # set cell ids to avoid:
145    # "MissingIDFieldWarning: Code cell is missing an id field, this will become a hard error in future nbformat versions."
146    header_cell["id"] = "wvpy_header_cell"
147    orig_cells = [c.copy() for c in nb.cells]
148    for i in range(len(orig_cells)):
149        orig_cells[i]["id"] = f"cell{i}"
150    cells = [header_cell] + orig_cells
151    nb_out = nbf_v.new_notebook(cells=cells)
152    # nb_out = _normalize(nb_out)
153    return nb_out

184def convert_notebook_code_to_py(
185    nb: nbformat.notebooknode.NotebookNode,
186    *,
187    use_black: bool = False,
188) -> str:
189    """
190    Convert ipython notebook inputs to a py code.
191    "''' begin text" ends any open blocks, and starts a new markdown block (triple double quotes also allowed)
192    "''' # end text" ends text, and starts a new code block (triple double quotes also allowed)
193    "'''end code'''" ends code blocks, and starts a new code block (triple double quotes also allowed)
194
195    :param nb: notebook
196    :param use_black: if True use black to re-format Python code
197    :return: Python source code
198    """
199    assert isinstance(use_black, bool)
200    res = []
201    code_needs_end = False
202    for cell in nb.cells:
203        if len(cell.source.strip()) > 0:
204            if cell.cell_type == "code":
205                if code_needs_end:
206                    res.append('\n"""end code"""\n')
207                py_text = cell.source.strip("\n") + "\n"
208                if use_black and have_black:
209                    py_text = pretty_format_python(py_text)
210                res.append(py_text)
211                code_needs_end = True
212            else:
213                res.append('\n""" begin text')
214                res.append(cell.source.strip("\n"))
215                res.append('"""  # end text\n')
216                code_needs_end = False
217    res_text = "\n" + ("\n".join(res)).strip("\n") + "\n\n"
218    return res_text

284def render_as_html(
285    notebook_file_name: str,
286    *,
287    output_suffix: Optional[str] = None,
288    timeout: int = 60000,
289    kernel_name: Optional[str] = None,
290    verbose: bool = True,
291    sheet_vars=None,
292    init_code: Optional[str] = None,
293    exclude_input: bool = False,
294    prompt_strip_regexp: Optional[
295        str
296    ] = r'<\s*div\s+class\s*=\s*"jp-OutputPrompt[^<>]*>[^<>]*Out[^<>]*<\s*/div\s*>',
297) -> None:
298    """
299    Render a Jupyter notebook in the current directory as HTML.
300    Exceptions raised in the rendering notebook are allowed to pass trough.
301
302    :param notebook_file_name: name of source file, must end with .ipynb or .py (or type gotten from file system)
303    :param output_suffix: optional name to add to result name
304    :param timeout: Maximum time in seconds each notebook cell is allowed to run.
305                    passed to nbconvert.preprocessors.ExecutePreprocessor.
306    :param kernel_name: Jupyter kernel to use. passed to nbconvert.preprocessors.ExecutePreprocessor.
307    :param verbose logical, if True print while running
308    :param sheet_vars: if not None value is de-serialized as a variable named "sheet_vars"
309    :param init_code: Python init code for first cell
310    :param exclude_input: if True, exclude input cells
311    :param prompt_strip_regexp: regexp to strip prompts, only used if exclude_input is True
312    :return: None
313    """
314    assert isinstance(notebook_file_name, str)
315    # deal with no suffix case
316    if (not notebook_file_name.endswith(".ipynb")) and (
317        not notebook_file_name.endswith(".py")
318    ):
319        py_name = notebook_file_name + ".py"
320        py_exists = os.path.exists(py_name)
321        ipynb_name = notebook_file_name + ".ipynb"
322        ipynb_exists = os.path.exists(ipynb_name)
323        if (py_exists + ipynb_exists) != 1:
324            raise ValueError(
325                "{ipynb_exists}: if file suffix is not specified then exactly one of .py or .ipynb file must exist"
326            )
327        if ipynb_exists:
328            notebook_file_name = notebook_file_name + ".ipynb"
329        else:
330            notebook_file_name = notebook_file_name + ".py"
331    # get the input
332    assert os.path.exists(notebook_file_name)
333    if notebook_file_name.endswith(".ipynb"):
334        suffix = ".ipynb"
335        with open(notebook_file_name, "rb") as f:
336            nb = nbformat.read(f, as_version=4)
337    elif notebook_file_name.endswith(".py"):
338        suffix = ".py"
339        with open(notebook_file_name, "r", encoding="utf-8") as f:
340            text = f.read()
341        nb = convert_py_code_to_notebook(text)
342    else:
343        raise ValueError("{ipynb_exists}: file must end with .py or .ipynb")
344    tmp_path = None
345    # do the conversion
346    if sheet_vars is not None:
347        with tempfile.NamedTemporaryFile(delete=False) as ntf:
348            tmp_path = ntf.name
349            pickle.dump(sheet_vars, file=ntf)
350        if (init_code is None) or (len(init_code) <= 0):
351            init_code = ""
352        pickle_code = f"""
353import pickle
354with open({tmp_path.__repr__()}, 'rb') as pf:
355   sheet_vars = pickle.load(pf)
356"""
357        init_code = init_code + "\n\n" + pickle_code
358    if (init_code is not None) and (len(init_code) > 0):
359        assert isinstance(init_code, str)
360        nb = prepend_code_cell_to_notebook(nb, code_text=f"\n\n{init_code}\n\n")
361    html_name = os.path.basename(notebook_file_name)
362    html_name = html_name.removesuffix(suffix)
363    exec_note = ""
364    if output_suffix is not None:
365        assert isinstance(output_suffix, str)
366        html_name = html_name + output_suffix
367        exec_note = f'"{output_suffix}"'
368    html_name = html_name + ".html"
369    try:
370        os.remove(html_name)
371    except FileNotFoundError:
372        pass
373    caught = None
374    trace = None
375    html_body = ""
376    if verbose:
377        print(
378            f'start render_as_html "{notebook_file_name}" {exec_note} {datetime.datetime.now()}'
379        )
380    try:
381        if kernel_name is not None:
382            ep = OurExecutor(timeout=timeout, kernel_name=kernel_name)
383        else:
384            ep = OurExecutor(timeout=timeout)
385        nb_res, nb_resources = ep.preprocess(nb)
386        html_exporter = HTMLExporter(exclude_input=exclude_input)
387        html_body, html_resources = html_exporter.from_notebook_node(nb_res)
388        caught = ep.caught_exception
389    except Exception as e:
390        caught = e
391        trace = traceback.format_exc()
392    if exclude_input and (prompt_strip_regexp is not None):
393        # strip output prompts
394        html_body = re.sub(prompt_strip_regexp, " ", html_body)
395    with open(html_name, "wt", encoding="utf-8") as f:
396        f.write(html_body)
397        if caught is not None:
398            f.write("\n<pre>\n")
399            f.write(escape_ansi(str(caught)))
400            f.write("\n</pre>\n")
401        if trace is not None:
402            f.write("\n<pre>\n")
403            f.write(escape_ansi(str(trace)))
404            f.write("\n</pre>\n")
405    nw = datetime.datetime.now()
406    if tmp_path is not None:
407        try:
408            os.remove(tmp_path)
409        except FileNotFoundError:
410            pass
411    if caught is not None:
412        if verbose:
413            print(
414                f'\n\n\texception in render_as_html "{notebook_file_name}" {nw} {escape_ansi(str(caught))}\n\n'
415            )
416            if trace is not None:
417                print(f"\n\n\t\ttrace {escape_ansi(str(trace))}\n\n")
418        raise caught
419    if verbose:
420        print(f'\tdone render_as_html "{notebook_file_name}" {nw}')

432@total_ordering
433class JTask:
434    def __init__(
435        self,
436        sheet_name: str,
437        *,
438        output_suffix: Optional[str] = None,
439        exclude_input: bool = True,
440        sheet_vars=None,
441        init_code: Optional[str] = None,
442        path_prefix: Optional[str] = None,
443        strict: bool = True,
444    ) -> None:
445        """
446        Create a Jupyter task.
447
448        :param sheet_name: name of sheet to run can be .ipynb or .py, and suffix can be omitted.
449        :param output_suffix: optional string to append to rendered HTML file name.
450        :param exclude_input: if True strip input cells out of HTML render.
451        :param sheet_vars: if not None value is de-serialized as a variable named "sheet_vars"
452        :param init_code: optional code to insert at the top of the Jupyter sheet, used to pass parameters.
453        :param path_prefix: optional prefix to add to sheet_name to find Jupyter source.
454        :param strict: if True check paths path_prefix and path_prefix/sheetname[.py|.ipynb] exist.
455        """
456        assert isinstance(sheet_name, str)
457        assert isinstance(output_suffix, (str, type(None)))
458        assert isinstance(exclude_input, bool)
459        assert isinstance(init_code, (str, type(None)))
460        assert isinstance(path_prefix, (str, type(None)))
461        if strict:
462            path = sheet_name
463            if (isinstance(path_prefix, str)) and (len(path_prefix) > 9):
464                assert os.path.exists(path_prefix)
465                path = os.path.join(path_prefix, sheet_name)
466            if path.endswith(".py"):
467                path = path.removesuffix(".py")
468            if path.endswith(".ipynb"):
469                path = path.removesuffix(".ipynb")
470            py_exists = os.path.exists(path + ".py")
471            ipynb_exists = os.path.exists(path + ".ipynb")
472            assert (py_exists + ipynb_exists) >= 1
473            if (not sheet_name.endswith(".py")) and (not sheet_name.endswith(".ipynb")):
474                # no suffix, so must be unambiguous
475                assert (py_exists + ipynb_exists) == 1
476        self.sheet_name = sheet_name
477        self.output_suffix = output_suffix
478        self.exclude_input = exclude_input
479        self.sheet_vars = sheet_vars
480        self.init_code = init_code
481        self.path_prefix = path_prefix
482
483    def render_as_html(self) -> None:
484        """
485        Render Jupyter notebook or Python (treated as notebook) to HTML.
486        """
487        path = self.sheet_name
488        if isinstance(self.path_prefix, str) and (len(self.path_prefix) > 0):
489            path = os.path.join(self.path_prefix, self.sheet_name)
490        render_as_html(
491            path,
492            exclude_input=self.exclude_input,
493            output_suffix=self.output_suffix,
494            sheet_vars=self.sheet_vars,
495            init_code=self.init_code,
496        )
497
498    def render_py_txt(self) -> None:
499        """
500        Render Python to text (without nbconvert, nbformat Jupyter bindings)
501        """
502        path = self.sheet_name
503        if isinstance(self.path_prefix, str) and (len(self.path_prefix) > 0):
504            path = os.path.join(self.path_prefix, self.sheet_name)
505        execute_py(
506            path,
507            output_suffix=self.output_suffix,
508            sheet_vars=self.sheet_vars,
509            init_code=self.init_code,
510        )
511
512    def __getitem__(self, item):
513        return getattr(self, item)
514
515    def _is_valid_operand(self, other):
516        return isinstance(other, JTask)
517
518    def __eq__(self, other):
519        if not self._is_valid_operand(other):
520            return False
521        if str(type(self)) != str(type(other)):
522            return False
523        for v in _jtask_comparison_attributes:
524            if self[v] != other[v]:
525                return False
526        if str(self.sheet_vars) != str(other.sheet_vars):
527            return False
528        return True
529
530    def __lt__(self, other):
531        if not self._is_valid_operand(other):
532            return NotImplemented
533        if str(type(self)) < str(type(other)):
534            return True
535        for v in _jtask_comparison_attributes:
536            v_self = self[v]
537            v_other = other[v]
538            # can't order compare None to None
539            if (v_self is None) or (v_other is None):
540                if (v_self is None) != (v_other is None):
541                    return v_self is None
542            else:
543                if self[v] < other[v]:
544                    return True
545        if str(self.sheet_vars) < str(other.sheet_vars):
546            return True
547        return False

549    def __str__(self) -> str:
550        args_str = ",\n".join(
551            [
552                f" {v}={repr(self[v])}"
553                for v in _jtask_comparison_attributes + ["sheet_vars"]
554            ]
555        )
556        return "JTask(\n" + args_str + ",\n)"
557
558    def __repr__(self) -> str:
559        return self.__str__()
